MainActivity.java 
package com.example.android.tennismatchtracker; 
 
import android.support.v7.app.AppCompatActivity; 
import android.os.Bundle; 
import android.view.View; 
import android.widget.TextView; 
 
 
public class MainActivity extends AppCompatActivity { 
 
 
    @Override 
    protected void onCreate(Bundle savedInstanceState) { 
        super.onCreate(savedInstanceState); 
        setContentView(R.layout.activity_main); 
    } 
 
    // tracks Player A Game and Set score 
    int gameA = 0; 
    int setA = 0; 
 
    // tracks Player B Game and Set score 
    int gameB = 0; 
    int setB = 0; 
 
    // tracks Player A Points score 
    int pointA = 0; 
 
    // tracks Player B Points score 
    int pointB = 0; 
    // tracks Tie Break or not 
    int setTieBreak = 0; 
    int deuce = 0; 
    String advantage = "Adv"; 
 
 
    // tracks servesA 
    int serveOutA = 0; 
    int serveOutB = 0; 
 
 
    // * This method is called when player A Serve Out Button is pressed. 
 
 
    public void OutServeA(View view) { 
        serveOutA = serveOutA + 1; 
        if (serveOutA == 2) { 
            incrementPointB(); 
        } 
 
    } 
 
    // * This method is called when player B Serve Out Button is pressed. 
 
 
    public void OutServeB(View view) { 
        serveOutB = serveOutB + 1; 
        if (serveOutB == 2) { 
            incrementPointA(); 
        } 
 
    } 
// * This method is called to increment Player A Points Score is pressed. 
 
    public void incrementPointA() { 
 
        serveOutA = 0; 
        serveOutB = 0; 
        if (setTieBreak == 1) { 
            pointA = pointA + 1; 
            displayPointA(pointA); 
            if (pointA >= 7 && (pointA - pointB) > 1) { 
                setA = setA + 1; 
                newSet(); 
            } 
 
        } else if (deuce == 1) { 
            pointA = pointA + 1; 
            if (pointA == 2) { 
                pointA = 0; 
                pointB = 0; 
                deuce = 0; 
                displayPointA(pointA); 
                displayPointB(pointB); 
                gameA = gameA + 1; 
                displayGameA(gameA); 
                if (gameA == 6 && gameB == 6) { 
                    setTieBreak = 1; 
                } else if (gameA >= 6 && (gameA - gameB) > 1) { 
                    setA = setA + 1; 
                    newSet(); 
                } 
            } else if (pointB == 1) { 
                pointA = 40; 
                pointB = 40; 
                displayPointA(pointA); 
                displayPointB(pointB); 
                pointA = 0; 
                pointB = 0; 
            } else { 
                pointB = 0; 
                displayPointA(pointA); 
                displayPointB(pointB); 
            } 
        } else { 
            pointA = pointA + 15; 
            if (pointA == 45) { 
                pointA = 40; 
                if (pointB == 40) { 
                    displayPointA(pointA); 
                    displayPointB(pointB); 
                    pointA = 0; 
                    pointB = 0; 
                    deuce = 1; 
                } else { 
                    displayPointA(pointA); 
 
                } 
            } else if (pointA < 55) { 
                displayPointA(pointA); 
                displayPointB(pointB); 
            } else { 
                pointA = 0; 
                pointB = 0; 
                displayPointA(pointA); 
                displayPointB(pointB); 
                gameA = gameA + 1; 
                displayGameA(gameA); 
                if (gameA == 6 && gameB == 6) { 
                    setTieBreak = 1; 
                } else if (gameA >= 6 && (gameA - gameB) > 1) { 
                    setA = setA + 1; 
                    newSet(); 
                } 
            } 
        } 
 
    } 
 
 
    // * This method is called when Player A Point Won Button is pressed. 
 
 
    public void scorePointA(View view) { 
 
        incrementPointA(); 
 
    } 
 
 
    // * This method is called to increment Player B Points Score is pressed. 
 
 
    public void incrementPointB() { 
 
        serveOutA = 0; 
        serveOutB = 0; 
        if (setTieBreak == 1) { 
            pointB = pointB + 1; 
            displayPointB(pointB); 
            if (pointB >= 7 && (pointB - pointA) > 1) { 
                setB = setB + 1; 
                newSet(); 
            } 
 
        } else if (deuce == 1) { 
            pointB = pointB + 1; 
            if (pointB == 2) { 
                pointA = 0; 
                pointB = 0; 
                deuce = 0; 
                displayPointA(pointA); 
                displayPointB(pointB); 
                gameB = gameB + 1; 
                displayGameB(gameB); 
                if (gameB == 6 && gameA == 6) { 
                    setTieBreak = 1; 
                } else if (gameB >= 6 && (gameB - gameA) > 1) { 
                    setB = setB + 1; 
                    newSet(); 
                } 
            } else if (pointA == 1) { 
                pointA = 40; 
                pointB = 40; 
                displayPointA(pointA); 
                displayPointB(pointB); 
                pointA = 0; 
                pointB = 0; 
            } else { 
                pointA = 0; 
                displayPointA(pointA); 
                displayPointB(pointB); 
            } 
        } else { 
            pointB = pointB + 15; 
            if (pointB == 45) { 
                pointB = 40; 
                if (pointA == 40) { 
                    displayPointA(pointA); 
                    displayPointB(pointB); 
                    pointA = 0; 
                    pointB = 0; 
                    deuce = 1; 
                } 
                else { 
                    displayPointB(pointB); 
                } 
            } else if (pointB < 55) { 
                        displayPointA(pointA); 
                        displayPointB(pointB); 
                    } else { 
                        pointB = 0; 
                        pointA = 0; 
                        displayPointB(pointB); 
                        displayPointA(pointA); 
                        gameB = gameB + 1; 
                        displayGameB(gameB); 
                        if (gameB == 6 && gameA == 6) { 
                            setTieBreak = 1; 
                        } else if (gameB >= 6 && (gameB - gameA) > 1) { 
                            setB = setB + 1; 
                            newSet(); 
                        } 
                    } 
                } 
        } 
 
 
 
    // * This method is called when Player B Point Won Button is pressed. 
 
 
    public void scorePointB(View view) { 
 
 
        incrementPointB(); 
 
    } 
 
    /** 
     * Displays Player A Points Score. 
     */ 
    public void displayPointA(int scoreA) { 
        TextView scoreView = (TextView) findViewById(R.id.player_a_point); 
        if (deuce == 1 && scoreA == 1) { 
            scoreView.setText("ADV"); 
        } else { 
            scoreView.setText(String.valueOf(scoreA)); 
        } 
 
    } 
 
    /** 
     * Displays Player B Points Score. 
     */ 
    public void displayPointB(int scoreB) { 
        TextView scoreView = (TextView) findViewById(R.id.player_b_point); 
        if (deuce == 1 && scoreB == 1) { 
            scoreView.setText("ADV"); 
        } else { 
            scoreView.setText(String.valueOf(scoreB)); 
        } 
 
    } 
 
    /** 
     * Displays Player A Games Score. 
     */ 
    public void displayGameA(int scoreA) { 
        TextView scoreView = (TextView) findViewById(R.id.player_a_game); 
        scoreView.setText(String.valueOf(scoreA)); 
 
    } 
 
    /** 
     * Displays Player B Games Score. 
     */ 
    public void displayGameB(int scoreB) { 
        TextView scoreView = (TextView) findViewById(R.id.player_b_game); 
        scoreView.setText(String.valueOf(scoreB)); 
 
    } 
 
    /** 
     * Displays Player A Set Score. 
     */ 
 
    public void displaySetA(int scoreA) { 
        TextView scoreView = (TextView) findViewById(R.id.player_a_set); 
        scoreView.setText(String.valueOf(scoreA)); 
 
    } 
 
    /** 
     * Displays Player B Set Score. 
     */ 
 
    public void displaySetB(int scoreB) { 
        TextView scoreView = (TextView) findViewById(R.id.player_b_set); 
        scoreView.setText(String.valueOf(scoreB)); 
 
 
    } 
 
    /** 
     * This method is called when a set is won. 
     */ 
 
    public void newSet() { 
 
        gameA = 0; 
        gameB = 0; 
        pointA = 0; 
        pointB = 0; 
        displaySetA(setA); 
        displaySetB(setB); 
        displayPointA(pointA); 
        displayPointB(pointB); 
        displayGameA(gameA); 
        displayGameB(gameB); 
        setTieBreak = 0; 
 
 
    } 
 
 
    /** 
     * This method is called when RESET button is pressed. 
     */ 
    public void resetScore(View view) { 
 
        setA = 0; 
        setB = 0; 
        gameA = 0; 
        gameB = 0; 
        pointA = 0; 
        pointB = 0; 
        displayPointA(pointA); 
        displayPointB(pointB); 
        displayGameA(gameA); 
        displayGameB(gameB); 
        displaySetA(setA); 
        displaySetB(setB); 
    } 
 
 
}
